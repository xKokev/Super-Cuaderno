Proxies 


        
          游꼛 Es un mecanismo que permite crear un objeto basado en un objeto literal inicial 
          游꼖 HANDLER es el manejador y da las reglas a seguir y usaremos SET 
          const persona = {
            nombre: "",
            apellido: "",
            edad: 0
          }
          const manejador = {
            set(obj, prop, valor) {
              if (Object.keys(obj).indexOf(prop) === -1) {
                return console.error(`La propiedad "${prop}" no existe en el objeto persona.`);
              }
              if (
                (prop === "nombre" || prop === "apellido") &&
                !(/^[A-Za-z칌침츼치칄칠칈칤칍칩칔칰칖칲\s]+$/g.test(valor))
              ) {
                return console.error(`La propiedad "${prop}"" s칩lo acepta letras y espacios en blanco`);
              }
              obj[prop] = valor;
            }
          }
          const jon = new Proxy(persona, manejador);
          jon.nombre = "Jon";
          jon.apellido = "MirCha";
          jon.edad = 35;
          jon.twitter = "@jonmircha";
          console.log(jon);
          // Proxy맡nombre: "Jon", apellido: "MirCha", edad: 35}
