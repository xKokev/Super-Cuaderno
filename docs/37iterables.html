Iterables & Iterators 


        
          Hay una version mas sencilla con los generators 
          ⚽ ITERABLE es el elemento que su contenido se puede recorrer
          🏀 ITERADOR es el que lo recorre 

          🏈 CASO 1 ARRAY
          const iterable = [1, 2, 3, 4, 5];
          const iterador = iterable[Symbol.iterator]();
          console.log(iterable);
          // (5) [1, 2, 3, 4, 5]
          //Accedemos al iterador del iterable
          console.log(iterador);
          // Array Iterator {}

          ⚾ CASO 2 SPRING 
          const iterable = "Hola Mundo";
          const iterador = iterable[Symbol.iterator]();
          console.log(iterable);
          // Hola Mundo 
          //Accedemos al iterador del iterable
          console.log(iterador);
          // Spring Iterator {}

          🥎 CASO 3 SET 
          const iterable = new Set([1, 2, 3, 3, 4, 5]);
          const iterador = iterable[Symbol.iterator]();
          console.log(iterable);
          //Accedemos al iterador del iterable
          // Set(5) {1, 2, 3, 4, 5}
          console.log(iterador);
          // SetIterator {1, 2, 3, 4, 5}
          
          🏐 CASO 4 MAP 
          const iterable = new Map([["nombre", "jon"], ["edad", 35]]);
          const iterador = iterable[Symbol.iterator]();
          console.log(iterable);
          //Accedemos al iterador del iterable
          // Map(2) {"nombre" => "jon", "edad" => 35}
          console.log(iterador);
          // MapIterator {"nombre" => "jon", "edad" => 35}
