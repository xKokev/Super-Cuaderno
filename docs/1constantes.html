Vaumoos 
            
              😏 Constante
                const PI = 3.1416;
                console.log(PI);
              // 3.1416

              🦊 Consola 
                a = "Rocky";
                console.log(a);
              // Rocky

              😎 Objeto 
                const objeto = {
                nombre: "Jon",
                edad: 35
                }
                console.log(objeto);
              {nombre: "Jon", edad: 35}

              🐱 Añadir elemento al objeto  
                objeto.correo = "jonmircha@gmail.com";
                console.log(objeto);
              {nombre: "Jon", edad: 35, correo: "jonmircha@gmail.com"}   

              💩 Array
                const colores = ["blanco", "negro", "azul"];
                console.log(colores);
              ["blanco", "negro", "azul"]

              🎲 Añadir elemento a un array 
                colores.push("anaranjado");
                console.log(colores);
              ["blanco", "negro", "azul", "anaranjado"]
