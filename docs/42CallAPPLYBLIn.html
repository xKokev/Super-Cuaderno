Call Apply Blind 


        
          Si bien this hace referencia al scope donde se encuentra con CALL APPLY BLIND podemos indicar donde se pueda aplicar 
          üóø En windows 
          console.log(this);
          // Window¬†{window: Window, self: Window, document: document, name: "", location: Location,¬†‚Ä¶}

          üóΩ En la funcion 
          this.lugar = "Contexto Global";
          function saludar(saludo, aQuien) {
            console.log(`${saludo} ${aQuien} desde el ${this.lugar}`);
          }
          // Hola kEnAi desde el Contexto Global

          üè∞ En el objeto 
          saludar("Hola", "kEnAi");
          const obj = {
            lugar: "Contexto Objeto"
          }

          üèü CALL lo que hace esto es: usa el objeto saludar pero en otro contexto mando a "llamar" / "call" a otro this y en el caso de abajo es obj 
          saludar.call(obj, "Hola", "Jon");
          // Hola Jon desde el Contexto Objeto
          saludar.call(null, "Hola", "Jon");
          // Hola Jon desde el Contexto Global
          saludar.call(this, "Hola", "Jon");
          // Hola Jon desde el Contexto Global

          üé° APPLY funciona como CALL pero ademas de pasar el contexto puedo transmitir otros parametros [si pondemos null nos regresamos al contexto global, si usamos this iremos a donde este ese objeto   ]
          saludar.apply(obj, ["Adios", "MirCha"]);
          // Adios MirCha desde el Contexto Objeto
          saludar.apply(null, ["Adios", "MirCha"]);
          // Adios MirCha desde el Contexto Global
          saludar.apply(this, ["Adios", "MirCha"]);
          // Adios MirCha desde el Contexto Global
          this.nombre = "Window";

          üóºBIND hace un enlace es como en la arrow function que no generea un nuevo scope para si misma sino usa el scope donde fue creada 
          const persona = {
            nombre: "Jon",
            saludar: function () {
              console.log(`Hola ${this.nombre}`)
            }
          }
          persona.saludar();
          // Hola Jon 
          
          üèù BIND 
          const otraPersona = {
            saludar: persona.saludar.bind(this)
          }
          otraPersona.saludar();
          // Hola Window
