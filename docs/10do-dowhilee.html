Do - Do While 


        
          Indeterminados - No sabemos el número de veces que se va a ejecutar el código

          🐸 Bucle while
          while(condición){
              //código a ejecutar
          }

              let contador = 0;
              while (contador < 10) {
                console.log("while " + contador);
                contador++;
              }

          🐔 Bucle do...while
          do{
              //código a ejecutar
          }while(condición)

              let contador = 0;
              do {
                console.log("do while " + contador);
                contador++;
              } while (contador < 10);
