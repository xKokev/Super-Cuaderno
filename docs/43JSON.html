            ü§ç PARSE analiza un JSON y lo convierte en un tipo que javascript valide un objeto arreglo boolean [si ya es un objeto valido en Js PARSE lo deja valido] 
            console.log("*** JSON.parse ***");
            // *** JSON.parse ***

            console.log(JSON.parse("{}"));
            // {} pero convertido en objeto 

            console.log(JSON.parse("[1,2,3]"));
            // (3)[1, 2, 3]
            console.log(JSON.parse("true"));
            // true [en azulito]
            console.log(JSON.parse("false"));
            // false [en azulito]
            console.log(JSON.parse("19"));
            // 19 [en azulito]
            console.log(JSON.parse('"Si soy un string escribeme con cadenas dobles y cadenas simples"'));
            // Si soy un string escribeme con cadenas dobles y cadenas simples
            console.log(JSON.parse("null"));
            // null 
            console.log(JSON.parse("undefined"));
            // error 
            console.log(JSON.parse('{ "x": 2, "y": 3 }'));
            // {"x":2,"y":3} [un objeto]

            üíú STRINGIFY hace lo contrario de PARSE convierte un objeto valido de JS a cadena de texto 
            console.log("*** JSON.stringify ***");
            // *** Ejemplo JSON ***
            console.log(JSON.stringify({}));
            // {} [no es un objeto es una cadena de texto un corchetes y ya en gris ]
            console.log(JSON.stringify([1, 2, 3]));
            // [1,2,3] [no es un array, es una cadena de texto un corchetes y ya en gris ]
            console.log(JSON.stringify(true));
            // true [en gris]
            console.log(JSON.stringify(false));
            // false [en gris]
            console.log(JSON.stringify(19));
            // 19 [en gris]
            console.log(JSON.stringify("No hace falta las comillas dobles y simples"));
            // No hace falta las comillas dobles y simples
            console.log(JSON.stringify(null));
            // null 
            console.log(JSON.stringify(undefined));
            // undefined [ahora si lo permite sin error ]
